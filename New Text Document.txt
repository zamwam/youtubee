import sys
from PyQt5.QtWidgets import QApplication, QWidget, QVBoxLayout, QAction, QMainWindow, QMenu
from PyQt5.QtWebEngineWidgets import QWebEngineView
from PyQt5.QtCore import QUrl

class YouTubeViewer(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("YouTube Viewer")
        self.setGeometry(100, 100, 800, 600)

        self.create_menu()
        self.create_webview()

    def create_menu(self):
        settings_menu = self.menuBar().addMenu("Settings")
        appearance_action = QAction("Appearance", self)
        behavior_action = QAction("Behavior", self)
        preferences_action = QAction("Preferences", self)

        appearance_action.triggered.connect(self.show_appearance_settings)
        behavior_action.triggered.connect(self.show_behavior_settings)
        preferences_action.triggered.connect(self.show_preferences)

        settings_menu.addAction(appearance_action)
        settings_menu.addAction(behavior_action)
        settings_menu.addAction(preferences_action)

    def create_webview(self):
        self.webview = QWebEngineView()
        self.webview.setUrl(QUrl("https://www.youtube.com/"))
        self.setCentralWidget(self.webview)

    def show_appearance_settings(self):
        print("Appearance settings")

    def show_behavior_settings(self):
        print("Behavior settings")

    def show_preferences(self):
        print("Preferences")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    viewer = YouTubeViewer()
    viewer.show()
    sys.exit(app.exec_())
